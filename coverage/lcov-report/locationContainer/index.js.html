export default class LocationContainer extends Component {
 
  constructor(props) {

    this.state = {
      city: 'unknown',
      state: 'unknown',
      country: 'unknown',
    };
  }
 
  componentDidMount() {
    var url = `https://maps.googleapis.com/maps/api/geocode/json?latlng=
    ${this.props.latitude},${this.props.longitude}&sensor=false`;
 
    fetch(url)
    .then((response) => response.json())
    .then((responseJson) => {
      console.log('responseJson',JSON.stringify(responseJson.results[0], null, 2));
      this.setState({city: responseJson.results[0].address_components[2].long_name});

      this.setState({country: responseJson.results[0].address_components[5].long_name});
    })
    .catch((error) => {
      console.error(error);
    });
  }
 
  render() {
    return (
      <Location
        city={this.state.city}
        state={this.state.state}
        country={this.state.country}
        onPress={this.props.onPress} />
    );
  }
}
